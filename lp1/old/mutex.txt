// Producer-Consumer Problem using Semaphore and Mutex in Java
import java.util.concurrent.Semaphore;

public class ProducerConsumer {
    // Buffer size
    static final int BUFFER_SIZE = 5;
    // Total items to produce/consume
    static final int TOTAL_ITEMS = 10;

    // Shared buffer
    int[] buffer = new int[BUFFER_SIZE];
    int in = 0, out = 0;

    // Semaphores
    Semaphore mutex = new Semaphore(1);           // acts as a lock
    Semaphore empty = new Semaphore(BUFFER_SIZE); // counts empty slots
    Semaphore full = new Semaphore(0);            // counts filled slots

    // Producer thread
    class Producer extends Thread {
        public void run() {
            int item = 0;
            try {
                while (item < TOTAL_ITEMS) { // produce limited items
                    item++; // produce item
                    empty.acquire();   // wait if buffer is full
                    mutex.acquire();   // lock the buffer

                    // Add item to buffer
                    buffer[in] = item;
                    System.out.println("Producer produced: " + item);
                    in = (in + 1) % BUFFER_SIZE;

                    mutex.release();   // unlock the buffer
                    full.release();    // increase count of full slots

                    Thread.sleep(800); // small delay
                }
            } catch (InterruptedException e) {
                System.out.println("Producer interrupted");
            }
            System.out.println("Producer finished producing all items.");
        }
    }

    // Consumer thread
    class Consumer extends Thread {
        public void run() {
            int count = 0;
            try {
                while (count < TOTAL_ITEMS) { // consume same number of items
                    full.acquire();    // wait if buffer is empty
                    mutex.acquire();   // lock the buffer

                    // Remove item from buffer
                    int item = buffer[out];
                    System.out.println("Consumer consumed: " + item);
                    out = (out + 1) % BUFFER_SIZE;

                    mutex.release();   // unlock buffer
                    empty.release();   // increase count of empty slots

                    count++;
                    Thread.sleep(1200); // small delay
                }
            } catch (InterruptedException e) {
                System.out.println("Consumer interrupted");
            }
            System.out.println("Consumer finished consuming all items.");
        }
    }

    public static void main(String[] args) {
        ProducerConsumer pc = new ProducerConsumer();
        Producer producer = pc.new Producer();
        Consumer consumer = pc.new Consumer();

        producer.start();
        consumer.start();
    }
}
